  (function () {
    var chips = document.querySelectorAll(".pub-filter");
    function apply(f) {
      chips.forEach(function (c) {
        c.classList.toggle("active", c.getAttribute("data-filter") === f);
      });
      document.querySelectorAll(".pub").forEach(function (p) {
        p.style.display = (f === "all" || p.getAttribute("data-topic") === f) ? "" : "none";
      });
      document.querySelectorAll(".pub-section").forEach(function (s) {
        var pubs = s.querySelectorAll(".pub");
        var visible = false;
        pubs.forEach(function (p) { if (p.style.display !== "none") visible = true; });
        s.style.display = visible ? "" : "none";
      });
    }
    chips.forEach(function (c) {
      c.addEventListener("click", function () { apply(c.getAttribute("data-filter")); });
    });
  })();
</script>
